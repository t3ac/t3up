## Leeren der History
DELETE FROM `sys_history` WHERE `history_data` IS NULL;
DELETE FROM `sys_history` WHERE `userid` IS NULL;
DELETE FROM `sys_history` WHERE `tablename` LIKE 'sys_file_reference';
DELETE FROM `sys_history` WHERE `tablename` LIKE 'sys_template';

DELETE FROM `sys_history` WHERE `tablename` LIKE 'fe_users';
DELETE FROM `sys_history` WHERE `tablename` LIKE 'be_users';


## Powermail
DELETE FROM `tx_powermail_domain_model_answer` WHERE deleted = 1;
DELETE FROM `tx_powermail_domain_model_mail` WHERE deleted = 1;

# Loeschen der Sysfileeintraege die nicht mehr existierenden Daten
DELETE FROM sys_file WHERE sys_file.missing=1 AND sys_file.uid NOT IN (SELECT sys_file_reference.uid_local FROM sys_file_reference WHERE sys_file_reference.uid_local=sys_file.uid); 

# Ruecksetzen auf text, wenn textpic keine Bilder hat
UPDATE `tt_content` SET `CType` = 'text' WHERE `CType` LIKE 'textpic' AND `image` = 0;

# Ruecksetzen auf text, wenn textmedia keine Bilder hat
UPDATE `tt_content` SET `CType` = 'text' WHERE `CType` LIKE 'textmedia' AND `assets` = 0;
